1 /*!*
  2  * Title: A responsive and performant content Slider
  3  * jQSlider follows a new approach in sliding content. Where most slider-plugins are moving the whole list of slides
  4  * when animating, jQSlider animates only the two slides necessary for the animation simultaneously. But the most
  5  * important difference is, that it is not cloning the first (respectively last) slides to realize a circular movement.
  6  * This is important if you have rich content inside of your slide that would break when it's cloned. This approach
  7  * makes it also possible to have a fully css based scaling.
  8  *
  9  * Usage:
 10  *
 11  * (start code)
 12  *  $("#my-jqslider").jqslider();
 13  * (end)
 14  *
 15  * See the jqslider documentation page for a full description.
 16  */
 17 
 18 /**
 19  * @fileOverview This is a jquery slider plugin
 20  * @name JQSlider
 21  * @date $Date: 2012-01-07 $
 22  * @author Marcello di Simone

 24  */

 26 /**
 27  * Closure Function to create a namespace for the initialization of the Plugin
 28  * @name anonymous
 29  * @function
 30  * @param {Object}       $            Aliased jQuery object.
 31  * @param {HTMLElement}  windows      Reference to the window object
 32  * @param {HTMLElement}  document     Reference to the document object
 33  * @param {undefined}    undefined    Reference to the document object
 34  */
 35 ;(function( $, window, document, undefined ){
 36 
 37      /**
 38       * jQuery definition to anchor JsDoc comments.
 39       *
 40       * @see http://jquery.com/
 41       * @name fn
 42       * @memberOf $
 43       * @namespace jQuery user methods container (plungin)
 44       * @ignore
 45       */
 46 
 47     /**
 48      * @class JQSlider Plugin Class
 49      * @name JQSlider
 50      * @constructor
 51      * @param {HTMLElement} elem    element to be initialized
 52      * @param {Object}    options   Options for the Plugin Member
 53      */
 54     function JQSlider( elem, options ){
 55         /**
 56          * Stores a reference to the jQuery object of the module
 57          * @name JQSlider#$el
 58          * @type Object
 59          */
 60         this.$el = $(elem);
 61         /**
 62          * Stores a reference to the document object
 63          * @name JQSlider#$doc
 64          * @type Object
 65          */
 66         this.$body = $('body');
 67         /**
 68          * This next line takes advantage of HTML5 data attributes
 69          * to support customization of the plugin on a per-element
 70          * basis (valid JSON only).
 71          * @name JQSlider#metadata
 72          * @type Object
 73          * @example
 74          * <div class="item" data-options='{"autosetup":"true"}'></div>
 75          */
 76         this.metadata = this.$el.data( 'options' );
 77         /**
 78          * Object with the configuration options of the module
 79          * @name JQSlider#o
 80          * @type Object
 81          */
 82         this.o = $.extend({}, this.defaults, options, this.metadata);
 83         /**
 84          * @name JQSlider#currentSlide
 85          * @type Number
 86          * @default 0
 87          */
 88         this.currentSlide = this.o.startSlide || 0;
 89         /**
 90          * @name JQSlider#isVertical
 91          * @type Boolean
 92          * @default false
 93          */
 94         this.isVertical = this.$el.hasClass( 'jqs-vertical' );
 95         /**
 96          * @name JQSlider#animating
 97          * @type Boolean
 98          * @default false
 99          */
100         this.animating = false;
101         /**
102          * @name JQSlider#slideTemplate
103          * @type String
104          * @default '<li/>'
105          */
106         this.slideTemplate = '<' + this.o.slideSelector + '/>';
107         /**
108          * alignment value shortcuts
109          * @name JQSlider#av
110          * @type Array
111          */
112         this.av = [
113             {pos:'left',size:'width'},
114             {pos:'top',size:'height'}
115         ];
116         /**
117          * reference of the slider container
118          * @name JQSlider#container
119          * @type jQuery
120          */
121         this.container = this.$el.children( this.o.containerSelector );
122         /**
123          * reference of the slider list element
124          * @name JQSlider#list
125          * @type jQuery
126          */
127         this.list = this.container.children( this.o.listSelector ).addClass( this.o.listClass );
128         /**
129          * jQuery set of all slide elements
130          * @name JQSlider#children
131          * @type jQuery
132          */
133         this.slides = this.list.children( this.o.slideSelector ).addClass( this.o.slideClass );
134 
135         if( this.o.autosetup !== false ) this.setup();
136     }
137 
138     /** @lends JQSlider.prototype */
139     JQSlider.prototype = {
140         /**
141          * Default configuration values
142          * @field
143          * @type Object
144          * @property {boolean} whether or not the setup should be called automatically
145          */
146         defaults: {
147             autosetup: true,
148             circular: false,
149             startSlide: 0,
150             animationSpeed:500,
151             easingFunction:'linear',
152             containerSelector:'.jqs-container',
153             listSelector:'ul',
154             listClass:'jqs-list',
155             slideSelector:'li',
156             slideClass:'jqs-slide'
157         },
158         /**
159          * Moves to the next slide
160          * @public
161          */
162         gotoNextSlide:function(){
163             var next = this._getNextSlideNumber();
164             // if the slider has no circular animation and the last slide is already present, do nothing
165             if( next !== false ){
166                 this.gotoSlide( next, false );
167             }
168         },
169         /**
170          * Moves to the previous slide
171          * @public
172          */
173         gotoPrevSlide:function(){
174             var prev = this._getPrevSlideNumber();
175             // if the slider has no circular animation and the first slide is already present, do nothing
176             if( prev !== false ) {
177                 this.gotoSlide( prev, true );
178             }
179         },
180         /**
181          * Moves to a designated slide
182          * @param {Number}  slide                   number of slide to go to
183          * @param {Boolean} [moveCounterwise=false]        set to true if animation should go to the left
184          * @param {Boolean} [noAnimation=false]     set to true if slide should be shown instantly without an animation
185          * @public
186          */
187         gotoSlide:function( slide, moveCounterwise, noAnimation ){
188             // stop if slider is currently animating or slide number is out of bound
189             if( this.animating === true || slide < 0 && slide >= this.slides.length ) return;
190 
191             this.animating = true;
192 
193             moveCounterwise = moveCounterwise || false;
194             
195             this.$el.trigger('animationstart', [ { current:this.currentSlide, next:slide, counterwise: moveCounterwise } ] );
196 
197             var self = this,
198                 next = $( this.slides[ slide ] ),
199                 current = $( this.slides[ this.currentSlide ] ),
200                 // extend currentCSS with the cssDefaults to avoid value pollution after orientation changes, which means
201                 // the top ,respectively left value, would be kept in the cssDefault object and cause a diagonal animation
202                 currentCSS = {},
203                 // here I typecast the boolean value this.isVertical to get the first or second index of the array av which holds the sting top or left
204                 elmPos = this.av[ +this.isVertical ].pos,
205                 // here I typecast the boolean value this.isVertical to get the first or second index of the array av which holds the sting width or height
206                 elmSize = this.av[ +this.isVertical ].size,
207                 moveSize = current[ elmSize ]();
208             if( noAnimation === true ){
209                 next.addClass('jqs-current');
210                 current.removeClass('jqs-current');
211                 this.currentSlide = slide;
212                 this.animating = false;
213                 this._resetControls();
214                 this.$el.trigger('animationend',[ { current:this.currentSlide, next:slide, counterwise: moveCounterwise } ]);
215             } else {
216                 // jQuery has a calculation bug in IE8 when translating negative percent values in pixels, therefor we set it ourself
217                 this.list.toggleClass( 'jqs-list-before', moveCounterwise ).css( elmPos, moveCounterwise? -moveSize:0 );
218                 next.addClass('jqs-next');
219                 currentCSS[ elmPos ]= ( moveCounterwise )? '0':-moveSize;
220                 // for debuging only
221                 //return
222                 this.list.animate( currentCSS, {
223                     duration: this.o.animationSpeed,
224                     easing: this.o.easingFunction,
225                     complete:  function(){
226                         current.removeClass('jqs-current');
227                         next.removeClass('jqs-next').addClass('jqs-current');
228                         self.list.attr('style','').removeClass( 'jqs-list-before' );
229                         self.currentSlide = slide;
230                         self.animating = false;
231                         self._resetControls();
232                         self.$el.trigger('animationend',[ { current:self.currentSlide, next:slide, counterwise: moveCounterwise } ]);
233                     }
234                     // for debuging only
235                     //,step:function(){window.console.debug('')}
236                 });
237             }
238         },
239         /**
240          * Returns the count of all slides
241          * @return {Number}     returns the number of all slides
242          * @public
243          */
244         getSlideCount:function(){
245             return this.slides.length;
246         },
247         /**
248          * Returns the position number of the current active slide
249          * @return {Number}     returns the index of the current slide in slides
250          * @see JQSlider#slides
251          * @public
252          */
253         getCurrentSlide:function(){
254             return this.currentSlide;
255         },
256         /**
257          * changes the orientation of the slider from horizontal to vertical or vertical to horizontal respectively
258          * @public
259          */
260         switchOrientation:function(){
261             this.$el.toggleClass( 'jqs-vertical' );
262             this.isVertical = !this.isVertical;
263         },
264         /**
265          * Adds a new slide node to the slider, if slidePosition is passed the new slide node will be added at a designated position
266          * @param {Number} [slidePosition]     position of the new slide to be added at
267          * @return {jQuery}                    returns the jquery object of the created slide
268          * @public
269          */
270         addSlide:function( slidePosition ){
271             var newSlide = $( this.slideTemplate );
272             if( undefined !== slidePosition && slidePosition < this.slides.length - 1 ){
273                 $( this.slides[ slidePosition ] ).before( newSlide );
274             } else {
275                 this.list.append( newSlide );
276             }
277             this.slides = this.list.children().addClass( this.o.slideClass );
278 
279             return newSlide;
280         },
281         /**
282          * Returns the position of the next slider
283          * @return {Number|Boolean}     returns index of next slide or false
284          * @private
285          */
286         _getNextSlideNumber:function(){
287             return ( ( this.currentSlide + 1 ) < this.slides.length )? ( this.currentSlide + 1 ):( this.o.circular )? 0:false;
288         },
289         /**
290          * Returns the position of the previous slider or false
291          * @return {Number|Boolean}     returns index of previous slide or false
292          * @private
293          */
294         _getPrevSlideNumber:function(){
295             return ( ( this.currentSlide - 1 ) >= 0 )? ( this.currentSlide - 1 ):( this.o.circular )? this.slides.length-1:false;
296         },
297         /**
298          * Hides the previousButton, respectively nextButton, if no circular animation is set and the first, respectively
299          * last slide is reached
300          * @private
301          */
302         _resetControls:function(){
303             if( this.buttons.length && !this.o.circular ){
304                 this.buttons.removeClass('jqs-inactive');
305                 if( this.currentSlide == 0 ) {
306                     this.prevButton.addClass('jqs-inactive');
307                 } else if( this.currentSlide == this.slides.length - 1 ){
308                     this.nextButton.addClass('jqs-inactive');
309                 }
310             }
311         },
312         /**
313          * Initialises the Slider controls and binds them to the previous and next methods
314          * @private
315          */
316         _initControls:function(){
317             var self = this;
318             this.buttons = this.$el.children('.jqs-handler');
319 
320             if( this.buttons.length ){
321                 this.nextButton = this.buttons.filter('.jqs-handler-next').bind('click',function(e){
322                     e.preventDefault();
323                     self.gotoNextSlide();
324                 });
325                 this.prevButton = this.buttons.filter('.jqs-handler-prev').bind('click',function(e){
326                     e.preventDefault();
327                     self.gotoPrevSlide();
328                 });
329                 if( !this.o.circular && this.currentSlide == 0 ) this.prevButton.addClass('jqs-inactive');
330             }
331         },
332         /**
333          * returns a random number to create an unique namespace, if no namespace is defined and no id is set on the root element
334          * @returns {Number}
335          * @private
336          */
337         _getUID:function () {
338             return (((1 + Math.random()) * 0x10000) | 0).toString( 16 ).substring( 1 );
339         },
340         /**
341          * the setup function
342          * @return JQSlider
343          * @public
344          */
345         setup:function(){
346             var self = this;
347             /**
348              * Namespace for the events that are triggered by self Instance
349              * this can also be used to have multiple slider with the same
350              * namespace so you're able to control same all at once with one event
351              * @name JQSlider#namespace
352              * @type String
353              */
354             this.nsSuffix = '.' + this.o.namespace || 'jqslider';
355 
356             $( this.slides[ this.currentSlide ] ).addClass('jqs-current');
357 
358             this._initControls();
359 
360             this.$el.addClass('jqs-initialiced');
361 
362             this.$el.bind( 'gotoprevslide' + this.nsSuffix, function( e ){ self.gotoPrevSlide(); } );
363             this.$el.bind( 'gotonextslide' + this.nsSuffix, function( e ){ self.gotoNextSlide(); } );
364             this.$el.bind( 'gotoslide' + this.nsSuffix, function( e, slideNum, counterwise, noAnimation ){ self.gotoSlide( slideNum, counterwise, noAnimation ); } );
365 
366             this.$el.trigger( 'init' );
367 
368             return this;
369         }
370     };
371     /**
372      * Expose the defaults options to the global scope (window) so you can change them for all modules at once
373      * @type    Object
374      */
375     JQSlider.defaults = JQSlider.prototype.defaults;
376     /**
377      * Initialize each object of the jQuery set as an instance of PluginBoilerplate
378      * @function
379      * @name jqslider
380      * @memberOf $.fn
381      * @param {Object} options  Object with plugin settings
382      * @return {Object} chainable jQuery object
383      */
384     $.fn.jqslider = function(options) {
385         return this.each(function(){
386             if ( undefined == $( this ).data('jqslider') ) {
387                 $( this ).data('jqslider', new JQSlider( this, options) );
388             }
389         });
390     };
391     window.JQSlider = JQSlider;
392 })( jQuery, window , document );
