1 /**
  2  * Title: A responsive and performant content Slider
  3  * jQSlider follows a new approach in sliding content. Where most slider-plugins are moving the whole list of slides
  4  * when animating, jQSlider animates only the two slides necessary for the animation simultaneously. But the most
  5  * important difference is, that it is not cloning the first (respectively last) slides to realize a circular movement.
  6  * This is important if you have rich content inside of your slide that would break when it's cloned. This approach
  7  * makes it also possible to have a fully css based scaling.
  8  *
  9  * Usage:
 10  *
 11  * (start code)
 12  *  $("#my-jqslider").jqslider();
 13  * (end)
 14  *
 15  * See the jqslider documentation page for a full description.
 16  *

 18  * and/or GPL (GPL-LICENSE.txt) licenses.
 19  *
 20  * Id: $Id: jquery.boilerplate.js 187 2011-11-27 13:03:28Z tom $
 21  */

 23 /**
 24  * @fileOverview This is a jquery slider plugin
 25  * @name JQSlider
 26  * @date $Date: 2012-01-07 $
 27  * @author <a href="mailto:mdisimone@t8y.com">Marcello di Simone</a>
 28  * @version 0.0.1
 29  */
 30 
 31 /**
 32  * Closure Function to create a namespace for the initialization of the Plugin
 33  * @name anonymous
 34  * @function
 35  * @param {object}      $           The name of this parameter aliased jQuery object. Here it is aliased to $.
 36  * @param {DOMElement}  windows     Reference to the window object
 37  * @param {DOMElement}  document    Reference to the document object
 38  * @param {undefined}   undefined   Reference to the document object
 39  */
 40 ;(function( $, window, document, undefined ){
 41     /**
 42      * global accessible reference to JQSlider (this)
 43      * @lends JQSlider
 44      * @type Boolean|JQSlider
 45      * @default false
 46      */
 47     var self = false;
 48     /**
 49      * @class JQSlider Plugin Class
 50      * @name JQSlider
 51      * @constructor
 52      * @param {DOMObject} elem      DOMElement to be initialized
 53      * @param {Object}    options   Options for the Plugin Member
 54      */
 55     var JQSlider = function( elem, options ){
 56         self = this;
 57         /**
 58          * Stores a reference to the DOMElement of the module
 59          * @name JQSlider#el
 60          * @type DOMObject
 61          */
 62         self.el = elem;
 63         /**
 64          * Stores a reference to the jQuery object of the module
 65          * @name JQSlider#$el
 66          * @type Object
 67          */
 68         self.$el = $(elem);
 69         /**
 70          * This next line takes advantage of HTML5 data attributes
 71          * to support customization of the plugin on a per-element
 72          * basis (valid JSON only).
 73          * @name JQSlider#metadata
 74          * @type Object
 75          * @example
 76          * <div class="item" data-options='{"autosetup":"true"}'></div>
 77          */
 78         self.metadata = self.$el.data( 'options' );
 79         /**
 80          * Object with the configuration options of the module
 81          * @name JQSlider#o
 82          * @type Object
 83          */
 84         self.o = $.extend({}, self.defaults, options, self.metadata);
 85         /**
 86          * @name JQSlider#currentSlide
 87          * @type Number
 88          * @default 0
 89          */
 90         self.currentSlide = self.o.startSlide || 0;
 91         /**
 92          * @name JQSlider#isVertical
 93          * @type Boolean
 94          * @default false
 95          */
 96         self.isVertical = self.$el.hasClass( 'jqslider-vertical' );
 97         /**
 98          * @name JQSlider#animating
 99          * @type Boolean
100          * @default false
101          */
102         self.animating = false;
103         /**
104          * alignment value shortcuts
105          * @name JQSlider#av
106          * @type Array
107          */
108         self.av = [
109             {pos:'left',size:'width'},
110             {pos:'top',size:'height'}
111         ];
112         /**
113          * set the configuration values for css3 transform plugin
114          * will be ignored if it's not included
115          * @name JQSlider#cssDefault
116          * @type Object
117          */
118         self.cssDefault = {
119             useTranslate3d: true,
120             leaveTransforms: false,
121             avoidTransforms: true
122         };
123         /**
124          * Namespace for the events that are triggered by self Instance
125          * this can also be used to have multiple slider with the same
126          * namespace so you're able to control same all at once with one event
127          * @name JQSlider#namespace
128          * @type String
129          */
130         self.namespace = self.$el.attr( 'id' ) || self.o.namespace || 'jqgrid-' + self._getUID();
131         /**
132          * reference of the slider container
133          * @name JQSlider#container
134          * @type jQuery
135          */
136         self.container = self.$el.children( self.o.containerSelector );
137         /**
138          * reference of the slider list element
139          * @name JQSlider#list
140          * @type jQuery
141          */
142         self.list = self.container.children( self.o.listSelector );
143         /**
144          * jQuery set of all slide elements
145          * @name JQSlider#children
146          * @type jQuery
147          */
148         self.children = self.list.children( self.o.slideSelector );
149         
150         if( self.o.autosetup !== false ) self.setup();
151     };
152 
153     /** @lends JQSlider.prototype */
154     JQSlider.prototype = {
155         /**
156          * Default configuration values
157          * @field
158          * @type Object
159          * @property {boolean} whether or not the setup should be called automatically
160          */
161         defaults: {
162             autosetup: true,
163             circular: false,
164             startSlide: 0,
165             animationSpeed:500,
166             easingFunction:'linear',
167             containerSelector:'.jqslider-container',
168             listSelector:'ul',
169             slideSelector:'li',
170             slideTemplate:'<li />'
171         },
172         /**
173          * Moves to the next slide
174          * @public
175          */
176         gotoNextSlide:function(){
177             var next = self._getNextSlideNumber();
178             // if the slider has no circular animation and the last slide is already present, do nothing
179             if( next !== false ){
180                 self.gotoSlide( next, false );
181             }
182         },
183         /**
184          * Moves to the previous slide
185          * @public
186          */
187         gotoPrevSlide:function(){
188             var prev = self._getPrevSlideNumber();
189             // if the slider has no circular animation and the first slide is already present, do nothing
190             if( prev !== false ) {
191                 self.gotoSlide( prev, true );
192             }
193         },
194         /**
195          * Moves to a designated slide
196          * @param {Number}  slide                   number of slide to go to
197          * @param {Boolean} [movePrev=false]        set to true if animation should go to the left
198          * @param {Boolean} [noAnimation=false]     set to true if slide should be shown instantly without an animation
199          * @public
200          */
201         gotoSlide:function( slide, movePrev, noAnimation ){
202             // stop if slider is currently animating or slide is out of bound
203             if( self.animating === true || slide < 0 && slide >= self.children.length ) return;
204 
205             self.animating = true;
206 
207             movePrev = movePrev || false;
208 
209             var next = $( self.children[ slide ] ),
210                 current = $( self.children[ self.currentSlide ] ),
211                 // extend currentCSS with the cssDefaults to avoid value pollution after orientation changes, which means
212                 // the top ,respectively left value, would be kept in the cssDefault object and cause a diagonal animation
213                 currentCSS = $.extend( {}, self.cssDefault );
214 
215             if( noAnimation === true ){
216                 next.addClass('jqslider-current');
217                 current.removeClass('jqslider-current');
218                 self.currentSlide = slide;
219                 self.animating = false;
220                 self._resetControls();
221             } else {
222                 self.list.toggleClass( 'jqslider-list-before', movePrev );
223                 next.addClass('jqslider-next');
224                 currentCSS[ self.av[ +self.isVertical ].pos ]= ( movePrev )? '0%':'-100%';
225                 self.list.animate( currentCSS, {
226                     duration: self.o.animationSpeed,
227                     easing: self.o.easingFunction,
228                     complete:  function(){
229                         current.removeClass('jqslider-current');
230                         next.removeClass('jqslider-next').addClass('jqslider-current');
231                         self.list.attr('style','').removeClass( 'jqslider-list-before' );
232                         self.currentSlide = slide;
233                         self.animating = false;
234                         self._resetControls();
235                     }
236                 });
237             }
238         },
239         /**
240          * Returns the count of all slides
241          * @return {Number}     returns the number of all slides
242          * @public
243          */
244 		getSlideCount:function(){
245 			return self.children.length;
246 		},
247         /**
248          * Returns the position number of the current active slide
249          * @return {Number}     returns the index of the current slide in children
250          * @see JQSlider#children
251          * @public
252          */
253 		getCurrentSlide:function(){
254 			return self.currentSlide;
255 		},
256         /**
257          * changes the orientation of the slider from horizontal to vertical or vertical to horizontal respectively
258          * @public
259          */
260         switchOrientation:function(){
261             self.$el.toggleClass( 'jqslider-vertical' );
262             self.isVertical = !self.isVertical;
263         },
264         /**
265          * Adds a new slide node to the slider, if slidePosition is passed the new slide node will be added at a designated position
266          * @param {Number} [slidePosition]     position of the new slide to be added at
267          * @return {jQuery}                    returns the jquery object of the created slide
268          * @public
269          */
270         addSlide:function( slidePosition ){
271             var newSlide = $( self.o.slideTemplate );
272             if( undefined !== slidePosition && slidePosition < self.children.length - 1 ){
273                 $( self.children[ slidePosition ] ).before( newSlide );
274             } else {
275                 self.list.append( newSlide );
276             }
277             self.children = self.list.children();
278 
279             return newSlide;
280         },
281         /**
282          * Returns the position of the next slider
283          * @return {Number|Boolean}     returns index of next slide or false
284          * @private
285          */
286         _getNextSlideNumber:function(){
287             return ( ( self.currentSlide + 1 ) < self.children.length )? ( self.currentSlide + 1 ):( self.o.circular )? 0:false;
288         },
289         /**
290          * Returns the position of the previous slider or false
291          * @return {Number|Boolean}     returns index of previous slide or false
292          * @private
293          */
294         _getPrevSlideNumber:function(){
295             return ( ( self.currentSlide - 1 ) >= 0 )? ( self.currentSlide - 1 ):( self.o.circular )? self.children.length-1:false;
296         },
297         /**
298          * Hides the previousButton, respectively nextButton, if no circular animation is set and the first, respectively
299          * last slide is reached
300          * @private
301          */
302         _resetControls:function(){
303             if( !self.o.circular ){
304                 self.buttons.removeClass('jqslider-inactive');
305                 if( self.currentSlide == 0 ) {
306                     self.prevButton.addClass('jqslider-inactive');
307                 } else if( self.currentSlide == self.children.length - 1 ){
308                     self.nextButton.addClass('jqslider-inactive');
309                 }
310             }
311         },
312         /**
313          * Initialises the Slider controls and binds them to the previous and next methods
314          * @private
315          */
316         _initControls:function(){
317         	self.buttons = self.$el.children('.jqslider-handler');
318             self.nextButton = self.buttons.filter('.jqslider-handler-next').bind('click',function(e){
319                 e.preventDefault();
320                 self.$el.trigger('gotonextslide.' + self.namespace);
321             });
322             self.prevButton = self.buttons.filter('.jqslider-handler-prev').bind('click',function(e){
323                 e.preventDefault();
324                 self.$el.trigger('gotoprevslide.' + self.namespace);
325             });
326 
327             if( !self.o.circular && self.currentSlide == 0 ) self.prevButton.addClass('jqslider-inactive');
328         },
329         /**
330          * returns a random number to create an unique namespace, if no namespace is defined and no id is set on the root element
331          * @returns {Number}
332          * @private
333          */
334         _getUID:function () {
335             return (((1 + Math.random()) * 0x10000) | 0).toString( 16 ).substring( 1 );
336         },
337         /**
338          * the setup function
339          * @return JQSlider
340          * @public
341          */
342         setup:function(){
343             $( self.children[ self.currentSlide ] ).addClass('jqslider-current');
344 
345             self._initControls();
346 
347             self.$el.bind('gotoprevslide.' + self.namespace, function( e ){ if( self.o.namespace && e.namespace != self.namespace ) return; self.gotoPrevSlide();});
348             self.$el.bind('gotonextslide.' + self.namespace, function( e ){ if( self.o.namespace && e.namespace != self.namespace ) return; self.gotoNextSlide();});
349 
350             self.$el.addClass('jqslider-initialiced');
351 
352             self.$el.trigger('init.' + self.namespace );
353             
354             return self;
355         }
356     };
357     /**
358      * Expose the defaults options to the global scope (window) so you can change them for all modules at once
359      * @see     window.JQSlider
360      * @scope   window
361      * @type    Object
362      */
363     JQSlider.defaults = JQSlider.prototype.defaults;
364     /**
365      * jQuery object's methods (This comment is just a JSDoc helper)
366      * @name fn
367      * @memberOf $
368      * @see anonymous
369      * @namespace jQuery user methods container (plugin)
370      * @ignore
371      */
372 
373     /**
374      * Initialize each object of the jQuery set as an instance of PluginBoilerplate
375      * @function
376      * @name pluginboilerplate
377      * @memberOf $.fn
378      * @return {jQuery} chainable jQuery class
379      */
380     $.fn.jqslider = function(options) {
381         return this.each(function(){
382             if ( undefined == $( this ).data('jqslider') ) {
383                 $( this ).data('jqslider', new JQSlider( this, options) );
384             }
385         });
386     };
387     window.JQSlider = JQSlider;
388 })( jQuery, window , document );
